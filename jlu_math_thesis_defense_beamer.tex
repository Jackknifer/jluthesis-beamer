\section{研究背景与问题提出}

\begin{frame}{研究背景}
  \begin{block}{本页应该写什么}
    对应论文第 1 章或第 1 节的研究背景。用 3--4 句话说明研究问题从哪里来、为什么值得研究、与实际场景有什么关系。
  \end{block}

  \begin{itemize}
    \item 背景来源：某类复杂系统的数据规模持续增长，传统分析流程难以快速响应。
    \item 现实问题：数据质量不稳定、变量关系复杂，人工判断成本较高。
    \item 研究必要性：需要构建更稳健、更易解释的分析框架来支撑后续决策。
  \end{itemize}
\end{frame}

\begin{frame}{研究现状与问题}
  \begin{block}{本页应该写什么}
    对应论文第 1 章或第 1 节的国内外研究现状。不要堆文献，按研究路线概括已有工作，并自然引出本文要解决的问题。
  \end{block}

  \begin{itemize}
    \item 研究路线一：基于规则或统计假设的方法结构清晰，便于解释。
    \item 研究路线二：基于数据驱动的方法适应性更强，适合复杂场景。
    \item 待解决问题：概括现有研究仍存在的不足，并引出本文切入点。
  \end{itemize}
\end{frame}

\begin{frame}{研究内容与贡献}
  \begin{block}{本页应该写什么}
    对应论文第 1 章或第 1 节的本文研究内容。建议只保留 3 条主贡献，每条都写成“做了什么 + 解决什么问题”。
  \end{block}

  \begin{enumerate}
    \item 研究内容一：提出一个面向目标任务的整体分析框架。
    \item 研究内容二：通过多组实验验证框架的稳定性和有效性。
    \item 研究内容三：结合一个实际案例展示方法的应用价值。
  \end{enumerate}
\end{frame}

\section{预备知识}

\begin{frame}{问题描述}
  \begin{block}{本页应该写什么}
    对应论文第 2 章或第 2 节的问题描述。本页要定义研究对象、输入输出、核心假设和目标任务。
  \end{block}

  \begin{equation*}
    \text{输入对象}
    \quad \longrightarrow \quad
    \text{模型或方法}
    \quad \longrightarrow \quad
    \text{目标输出}
  \end{equation*}

  \begin{itemize}
    \item 研究对象：一组随时间或样本变化而产生的多维观测数据。
    \item 基本假设：样本经过统一预处理，关键变量可以用于后续建模。
    \item 研究目标：在统一评价口径下得到稳定、准确且可解释的输出。
  \end{itemize}
\end{frame}

\begin{frame}{基本概念与评价指标}
  \begin{block}{本页应该写什么}
    对应论文第 2 章或第 2 节中的基本概念与评价指标。这里不需要长推导，重点解释后续实验表格和图像应该怎么看。
  \end{block}

  \begin{itemize}
    \item 基本概念：先解释“样本、特征、模型输出、评价指标”等核心术语。
    \item 评价指标一：准确性指标，用于衡量方法输出是否接近目标。
    \item 评价指标二：稳定性指标，用于衡量方法在不同场景下是否可靠。
    \item 对比原则：说明后续实验如何在统一口径下比较不同方法。
  \end{itemize}
\end{frame}

\section{模型与方法}

\begin{frame}{方法总体框架}
  \begin{block}{本页应该写什么}
    对应论文第 3 章或第 3 节的模型总述。建议放一张流程图，让老师先看到本文方法的整体逻辑。
  \end{block}

  \JLUFigurePlaceholder{建议放方法流程图：输入数据 -> 核心模型/算法 -> 关键统计量或输出 -> 判定/评价结果}

  \begin{itemize}
    \item 阶段一：整理原始数据，完成清洗、划分和特征准备。
    \item 阶段二：构建核心模型，并计算用于判断的关键输出。
    \item 阶段三：根据评价标准分析结果，并给出解释。
  \end{itemize}
\end{frame}

\begin{frame}{核心模型或算法}
  \begin{block}{本页应该写什么}
    对应论文第 3 章或第 3 节的核心模型。这里只讲最关键的设计，避免把实现细节、参数细节全部放进来。
  \end{block}

  \begin{equation*}
    \text{核心输入}
    \quad \xrightarrow{\ \text{本文方法}\ }
    \quad
    \text{核心输出}
  \end{equation*}

  \begin{itemize}
    \item 输入：经过预处理后的样本特征和必要的辅助信息。
    \item 输出：模型给出的评分、类别、统计量或综合判断。
    \item 设计理由：说明为什么该模型能解决前面提出的关键困难。
  \end{itemize}
\end{frame}

\begin{frame}{关键公式或统计量}
  \begin{block}{本页应该写什么}
    对应论文第 3 章或第 3 节的核心公式。只放最能代表本文方法的一到两个公式，并解释符号含义。
  \end{block}

  \begin{equation*}
    T = \Phi(\text{输入},\text{参数},\text{历史信息})
  \end{equation*}

  \begin{itemize}
    \item $T$：表示用于评价或判定的核心指标。
    \item 参数含义：解释公式里最重要的 2--3 个符号。
    \item 使用方式：说明该公式如何进入算法流程或结果判断。
  \end{itemize}
\end{frame}

\begin{frame}{参数设置与算法步骤}
  \begin{block}{本页应该写什么}
    对应论文第 3 章或第 3 节的算法流程。建议放算法表、流程图或 3--4 步文字，突出可复现的操作步骤。
  \end{block}

  \begin{enumerate}
    \item 步骤一：设定数据划分方式和主要参数。
    \item 步骤二：运行核心模型，得到中间统计量或预测结果。
