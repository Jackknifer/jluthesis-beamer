% 绘制封面/尾页的浅色答辩信息块。
% 第一个参数是蓝色块底边，第二个参数是信息块底边。
% 主体使用 JLUBlue!10；第二个 fill 用页面底色切出一条细边，形成嵌入蓝色块的留白。
\newcommand{\JLUDrawInfoBand}[2]{%
  \setlength{\JLUCoverInfoBandTopY}{\dimexpr#1+\JLUCoverInfoInsetDepth\relax}%
  \setlength{\JLUCoverInfoBandBottomY}{#2}%
  \setlength{\JLUCoverInfoBandHeight}{\dimexpr\JLUCoverInfoBandTopY-\JLUCoverInfoBandBottomY\relax}%
  \setlength{\JLUCoverInfoInsetRun}{.5\JLUCoverInfoInsetDepth}%
  \setlength{\JLUCoverInfoBottomRun}{.5\JLUCoverInfoBandHeight}%
  \fill[JLUBlue!10]
    (0,\JLUCoverInfoBandTopY)
    -- (\JLUCoverInfoTopWidth,\JLUCoverInfoBandTopY)
    -- (\dimexpr\JLUCoverInfoTopWidth+\JLUCoverInfoBottomRun\relax,\JLUCoverInfoBandBottomY)
    -- (0,\JLUCoverInfoBandBottomY)
    -- cycle;
  \fill[JLULightBlue]
    (0,\JLUCoverInfoBandTopY)
    -- (\dimexpr\JLUCoverInfoTopWidth+\JLUCoverInfoEdgeGap\relax,\JLUCoverInfoBandTopY)
    -- ++ (\JLUCoverInfoInsetRun,-\JLUCoverInfoInsetDepth)
    -- ++ (-\JLUCoverInfoEdgeGap,0)
    -- ++ (\dimexpr-\JLUCoverInfoInsetRun-.5\JLUCoverInfoEdgeGap\relax,\dimexpr\JLUCoverInfoInsetDepth-\JLUCoverInfoEdgeGap\relax)
    -- (0,\dimexpr\JLUCoverInfoBandTopY-\JLUCoverInfoEdgeGap\relax)
    -- cycle;
}

\section{\JLUSectionTitleOne}

\begin{frame}{关于本模板}
  \begin{block}{模板定位}
    本文件是吉林大学数学学院毕业论文答辩 Beamer 模板示例。正文内容为占位演示，使用时请替换为自己的论文内容。
  \end{block}

  \begin{itemize}
    \item 编译方式：使用 XeLaTeX 编译。
    \item 替换位置：论文信息、章节标题、正文 frame、图表和参考文献。
  \end{itemize}
\end{frame}

\begin{frame}{快速替换信息}
  \begin{block}{封面信息}
    请在文件开头的论文信息区替换题目、答辩人、学号、指导教师、专业和答辩日期。
  \end{block}

  \begin{itemize}
    \item \JLUEmph{题目}：建议控制在两行以内。
    \item \JLUEmph{副标题}：用于封面右侧、目录页左上角和页脚第三栏，默认写作“毕业论文答辩”。
    \item \JLUEmph{正文}：建议每页只保留一个核心观点，避免从论文中整段复制。
  \end{itemize}
\end{frame}

\begin{frame}{推荐答辩结构}
  \begin{block}{常见组织方式}
    如果没有特殊要求，可以按“背景问题、理论方法、实验结果、案例分析、总结展望”的顺序组织。
  \end{block}

  \begin{enumerate}
    \item 引出研究问题：说明问题来源、研究意义和主要困难。
    \item 展示核心工作：说明主要方法、关键公式或主要模型。
    \item 证明工作有效：用图表展示实验、案例和结论。
  \end{enumerate}
\end{frame}

    F(x)=\int_{a}^{b} f(t,x)\,\mathrm{d}t,

    \item $F(x)$ 表示目标函数、统计量或评价指标。
