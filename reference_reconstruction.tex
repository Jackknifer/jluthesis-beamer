\section{理论基础与方法}

\begin{frame}{核心问题定义}
  \vspace{.6cm}
  \begin{block}{问题定义}
    给定输入数据 $X=\{x_i\}_{i=1}^{n}$ 与目标变量 $Y=\{y_i\}_{i=1}^{n}$，
    本文希望学习一个映射函数 $f_{\theta}$，使其在目标任务上取得更好的性能。
  \end{block}

  \begin{equation}
    \label{eq:objective}
    \min_{\theta}\mathcal{L}(\theta)
    =
    \frac{1}{n}\sum_{i=1}^{n}
    \ell\left(f_{\theta}(x_i), y_i\right)
    + \lambda \lVert \theta \rVert_2^2 .
  \end{equation}

  式 \eqref{eq:objective} 中，$\lambda$ 表示正则化系数。
\end{frame}

\begin{frame}{研究方法}
  \vspace{.4cm}
  \begin{block}{方法概述}
    可以从数据处理、模型构建、算法优化和实验验证四个层面组织本页内容。
  \end{block}

  \begin{enumerate}
    \item 数据预处理：清洗、标注、特征构造或样本划分。
    \item 模型设计：核心结构、关键模块或数学假设。
    \item 优化策略：损失函数、训练流程或参数设置。
    \item 结果验证：指标选择、对比实验与消融分析。
  \end{enumerate}
\end{frame}

  \begin{lstlisting}[language=Python]
def train_epoch(model, loader, optimizer, criterion):
    model.train()

    for batch_x, batch_y in loader:
        optimizer.zero_grad()
        prediction = model(batch_x)
        loss = criterion(prediction, batch_y)
        loss.backward()
        optimizer.step()

    return loss.item()
  \end{lstlisting}
